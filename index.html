        let _scanResult = null;


        async function scanFoodWithClaude(base64, mediaType, apiKey) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`;

            

            // Explicit instructions for the AI to return ONLY JSON

            const prompt = "Act as a nutrition expert. Return ONLY a JSON object: {\"name\":\"Food Name\",\"description\":\"desc\",\"kcal\":0,\"protein\":0,\"carbs\":0,\"fat\":0}.";


            const body = {

                contents: [{

                    parts: [{ text: prompt }, { inline_data: { mime_type: mediaType, data: base64 } }]

                }]

            };


            try {

                const res = await fetch(url, { method: 'POST', body: JSON.stringify(body) });

                const data = await res.json();

                const rawText = data.candidates[0].content.parts[0].text;


                // CLEANING: Strip markdown backticks if AI adds them

                let jsonStr = rawText.replace(/```json/g, "").replace(/```/g, "").trim();

                const parsed = JSON.parse(jsonStr);


                _scanResult = {

                    name: parsed.name,

                    kcal: Math.round(parsed.kcal),

                    p: Math.round(parsed.protein),

                    c: Math.round(parsed.carbs),

                    f: Math.round(parsed.fat)

                };


                displayResult(_scanResult);

            } catch (err) {

                console.error("AI Error:", err);

                alert("Scanner failed. Check your API key or photo clarity.");

            }

        }


        // Handle Image Selection and Compression

        async function handlePhotoSelected(input) {

            const file = input.files[0];

            const key = localStorage.getItem('fitforge_api_key');

            if (!key) return alert("Please save your API key first.");


            const reader = new FileReader();

            reader.onload = async (e) => {

                const base64 = e.target.result.split(',')[1];

                document.getElementById('scanStatusLoading').style.display = 'block';

                await scanFoodWithClaude(base64, file.type, key);

            };

            reader.readAsDataURL(file);

        }


        function saveApiKey() {

            const val = document.getElementById('apikeyInput').value;

            if (val) {

                localStorage.setItem('fitforge_api_key', val);

                alert("Key Saved!");

            }

        }


        function displayResult(r) {

            document.getElementById('scanStatusLoading').style.display = 'none';

            document.getElementById('scanStatusResult').style.display = 'block';

            document.getElementById('srFoodName').textContent = r.name;

            document.getElementById('srMacros').textContent = `${r.kcal} kcal | P: ${r.p}g | C: ${r.c}g | F: ${r.f}g`;

        }
